#include <Adafruit_NeoPixel.h>//Ingreso
#include <Keypad.h>
#include <Servo.h>
#define Monitordetirmpo() delay(150)
#define FILAS 4
#define COLUMNAS 4

Adafruit_NeoPixel tira = Adafruit_NeoPixel(7,13);

char keys[FILAS][COLUMNAS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

char key;
byte Filas[FILAS]={9,8,7,6};
byte Columnas[COLUMNAS] = {5,4,3,2};

Keypad teclado = Keypad(makeKeymap(keys),Filas, Columnas,FILAS,COLUMNAS);

Servo servo;

void setup() {
  // put your setup code here, to run once:
  Serial.begin(9600);
  tira.begin();
  tira.show();
  servo.attach(10);

}

void loop() {

  key = teclado.getKey();

  if(key){
  Serial.println(key);
  }else{
    tira.clear();
  }
  switch(key){
    case '0':
      servo.write(0);
      Monitordetirmpo();
    break;

    case '1':
      servo.write(25);
      Monitordetirmpo();
    break;

    case '2':
      servo.write(45);
      Monitordetirmpo();
    break;

    case '3':
      servo.write(65);
      Monitordetirmpo();
    break;

    case '4':
      servo.write(85);
      Monitordetirmpo();
    break;

    case '5':
      servo.write(105);
      Monitordetirmpo();
    break;

    case '6':
      servo.write(125);
      Monitordetirmpo();
    break;

    case '7':
      servo.write(145);
      Monitordetirmpo();
    break;

    case '8':
      servo.write(165);
      Monitordetirmpo();
    break;

    case '9':
      servo.write(180);
      Monitordetirmpo();
    break;

    case 'A':
      for(int i = 0; i<7; i++) {
        tira.setBrightness(100);
        tira.setPixelColor(i, 100,201,90);
        tira.show();
       delay(100);
      }
    break;

    case 'B':
      for(int i = 0; i<7; i = i+2 ){
        tira.setBrightness(100);
        tira.setPixelColor(i, 10,230,200);
       tira.show();
        delay(100);
     }
    break;

    case 'C':
     for(int i = 0; i<7; i++) {
       tira.setBrightness(100);
        tira.setPixelColor(i, 225,90,70);
        tira.show();
       delay(100);
     } 
    break;

    case 'D':
      for(int i = 7; i>=-1; i = i-2) {
       tira.setBrightness(100);
        tira.setPixelColor(i, 10,10,255);
       tira.show();
       delay(100);
     }
    break;

    case '*' :
      for(int i=0;i<7 ;i++){
          tira.setBrightness(100);
          tira.setPixelColor(i, 25,10,150);
          tira.show();
          delay(100);
      }
    break;

    case '#':
      for(int i = 0; i<7; i = i+2 ){
       tira.setBrightness(100);
       tira.setPixelColor(i, 230,250,5);
       tira.show();
        delay(100);
      }
    break;
}

}
